# ── Núcleo de audio ──────────────────────────────────────────────────────────
sounddevice>=0.4.6          # captura de micrófono en tiempo real (reemplaza PyAudio)
numpy>=1.24                 # manipulación de arrays de audio

# ── VAD (Voice Activity Detection) ───────────────────────────────────────────
# Silero-VAD se descarga via torch.hub en tiempo de ejecución (~2 MB).
# Solo necesitamos torch como backend ligero (CPU):
torch>=2.0,<3.0             # backend de Silero-VAD (instalar versión CPU para ahorrar espacio)
torchaudio>=2.0,<3.0        # requerido por silero-vad

# ── STT (Speech-to-Text) ─────────────────────────────────────────────────────
faster-whisper>=1.0.0       # Whisper optimizado con CTranslate2; modelo "tiny" (~75 MB)
# El modelo se descarga automáticamente la primera vez que se instancia WhisperModel.

# ── Texto a voz (TTS) ────────────────────────────────────────────────────────
pyttsx3>=2.90               # TTS offline multiplataforma

# ── IA local (llama.cpp) ─────────────────────────────────────────────────────
llama-cpp-python            # requiere compilador C++; ver README para instrucciones

# ── Skills / utilidades ──────────────────────────────────────────────────────
requests>=2.31              # peticiones HTTP (clima, web, etc.)

# ────────────────────────────────────────────────────────────────────────────
# NOTAS DE INSTALACIÓN
# ────────────────────────────────────────────────────────────────────────────
#
# 1. PyTorch CPU-only (mucho más liviano, ~250 MB vs ~2 GB con CUDA):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/cpu
#    Luego: pip install -r requirements.txt
#
# 2. Si tienes GPU NVIDIA y quieres usarla (Faster-Whisper la detecta solo):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu121
#
# 3. sounddevice necesita libportaudio en Linux:
#    sudo apt install libportaudio2
#
# 4. Los modelos se descargan automáticamente al primer arranque:
#    - Silero-VAD   → ~/.cache/torch/hub/  (~2 MB)
#    - Whisper tiny → ~/.cache/huggingface/ (~75 MB)
#
# 5. SpeechRecognition y PyAudio ya NO son necesarios y pueden desinstalarse:
#    pip uninstall SpeechRecognition PyAudio
